üìä Supabase Schema Setup Required:

1. users table: 
CREATE TABLE users (
  id SERIAL PRIMARY KEY,
  pin TEXT UNIQUE NOT NULL,
  name TEXT NOT NULL,
  is_admin BOOLEAN DEFAULT FALSE,
  total_points INTEGER DEFAULT 0,
  created_at TIMESTAMP DEFAULT NOW()
);
2. betting_weeks table: 
CREATE TABLE betting_weeks (
  id SERIAL PRIMARY KEY,
  week_number INTEGER NOT NULL,
  year INTEGER NOT NULL,
  api_data JSONB,
  betting_deadline TIMESTAMP,
  is_active BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW()
);
3. user_bets table: 
CREATE TABLE user_bets (
  id SERIAL PRIMARY KEY,
  user_pin TEXT REFERENCES users(pin),
  week_id INTEGER REFERENCES betting_weeks(id),
  game_id TEXT NOT NULL,
  team_picked TEXT NOT NULL,
  bet_amount NUMERIC(10,2) NOT NULL,
  odds TEXT,
  is_locked BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMP DEFAULT NOW()
);
4. weekly_points table: 
CREATE TABLE weekly_points (
  id SERIAL PRIMARY KEY,
  user_pin TEXT REFERENCES users(pin),
  week_id INTEGER REFERENCES betting_weeks(id),
  points INTEGER NOT NULL,
  awarded_at TIMESTAMP DEFAULT NOW()
);
